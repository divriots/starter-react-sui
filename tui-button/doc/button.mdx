## Usage

```tsx
import { Button } from '../index';
import { theme } from '~/all';

export const Buttons = () => (
  <ThemeProvider theme={theme}>
    <Button mr={2}>Beep</Button>
    <Button mr={2} bg="divriots">
      Boop
    </Button>
    <Button mr={2} disabled>
      Disabled
    </Button>
  </ThemeProvider>
);
```
